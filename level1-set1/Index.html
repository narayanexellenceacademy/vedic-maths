📊 Google Sheets Setup Guide

Follow these steps to connect this test to your Google Sheet:

  1. Go to Google Sheets and create a new sheet. Name the first sheet tab: Test Results
  2. Add these headers in Row 1: Timestamp | Name | Mobile | Score | Percentage | Time Taken | Correct | Wrong | Unattempted
  3. Click Extensions → Apps Script
  4. Replace all code with the script below and click Save
  5. Click Deploy → New Deployment → Web App
  6. Set "Execute as" = Me and "Who has access" = Anyone
  7. Copy the Web App URL and paste it in the Admin setup box above
function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("Test Results"); var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.timestamp, data.name, data.mobile, data.score, data.percentage, data.timeTaken, data.correct, data.wrong, data.unattempted ]); return ContentService.createTextOutput( JSON.stringify({status:"success"}) ).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput( JSON.stringify({status:"error",message:err.toString()}) ).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { if(e.parameter && e.parameter.name){ var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("Test Results"); sheet.appendRow([ e.parameter.timestamp, e.parameter.name, e.parameter.mobile, e.parameter.score, e.parameter.percentage, e.parameter.timeTaken, e.parameter.correct, e.parameter.wrong, e.parameter.unattempted ]); } return ContentService.createTextOutput( JSON.stringify({status:"success"}) ).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput( JSON.stringify({status:"error",message:err.toString()}) ).setMimeType(ContentService.MimeType.JSON); } }
